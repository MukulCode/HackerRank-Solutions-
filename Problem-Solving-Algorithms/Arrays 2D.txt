#include <bits/stdc++.h>
using namespace std;

int main() {
    int arr[6][6],sum=0,max=INT_MIN;
    for (int i=0;i<6;i++){
        for(int j=0;j<6;j++){
            cin>>arr[i][j];
        }
    }
    
    for (int i=0;i<5;i++){
        for(int j=0;j<5;j++){
            if(i!=0 && j!=0){
                sum=arr[i-1][j-1]+arr[i-1][j]+arr[i-1][j+1]+arr[i][j]+arr[i+1][j-1]+arr[i+1][j]+arr[i+1][j+1];
                
                if(sum>max) max=sum;
                
            }
        }
    }
    
    
    cout<<max;
    
    return 0;
}
